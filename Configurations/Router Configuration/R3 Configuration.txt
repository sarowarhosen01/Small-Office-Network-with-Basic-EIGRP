enable
configure terminal
hostname R3

! Disable DNS lookup
no ip domain-lookup

! Configure interfaces
interface g0/0
ip address 10.1.23.2 255.255.255.252
no shutdown

interface g0/1
ip address 192.168.3.1 255.255.255.0
no shutdown

! Enable EIGRP
router eigrp 100
no auto-summary
network 192.168.3.0
network 10.1.23.0

end
write memory
