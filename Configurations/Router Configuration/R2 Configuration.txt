enable
configure terminal
hostname R2

! Disable DNS lookup
no ip domain-lookup

! Configure interfaces
interface gig0/0
ip address 10.1.12.2 255.255.255.252
no shutdown

interface g0/1
ip address 10.1.23.1 255.255.255.252
no shutdown

interface fa0/0/0
ip address 192.168.2.1 255.255.255.0
no shutdown

! Enable EIGRP
router eigrp 100
no auto-summary
network 192.168.2.0
network 10.1.12.0
network 10.1.23.0

end
write memory
