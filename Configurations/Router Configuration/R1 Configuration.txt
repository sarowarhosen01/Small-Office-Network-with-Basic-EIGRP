enable
configure terminal
hostname R1

! Disable DNS lookup (optional but helpful)
no ip domain-lookup

! Configure interfaces
interface gig0/1
ip address 192.168.1.1 255.255.255.0
no shutdown

interface gig0/0
ip address 10.1.12.1 255.255.255.252
no shutdown

! Enable EIGRP
router eigrp 100
! Disable auto-summary (critical for modern networks)
no auto-summary
! Advertise networks
network 192.168.1.0
network 10.1.12.0

! Save configuration
end
write memory
